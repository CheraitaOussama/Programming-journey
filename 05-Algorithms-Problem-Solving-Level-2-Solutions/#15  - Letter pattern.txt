#include<iostream>
using namespace std;

int ReadPositiveNumber(string Message) {
	int Number = 0;
	do
	{
		cout << Message << endl;
		cin >> Number;
	} while (Number<=0);
	return Number;
}
void PrintLetterPattern(int Number) {
	for (int i = 1; i <= Number; i++)
	{
		for (int j = 1; j <= i; j++)
		{
			cout << char(64+i);
		}
		cout << endl;
	}
}
int main() {
	PrintLetterPattern(ReadPositiveNumber("Enter a positive number? "));
	return 0;
}
