#include<iostream>
#include<stdlib.h>
using namespace std;
short RandomNumber(short From, short To) {
	return rand() % (To - From + 1) + From;
}
void FillArrayWithRandomNumbers(short Array[100], short&ArrayLength) {

	cout << "\nEnter Number of elements\n";
	cin >> ArrayLength;
	for (short  i = 0; i < ArrayLength; i++)
	{
		Array[i]= RandomNumber(1,100);
	}
}
void PrintArray(short Array[100], short ArrayLength) {

	for (short i = 0; i < ArrayLength; i++)
	{
		cout << Array[i] << " ";
	}
	cout << endl;
}
short MinNumberInArray(short Array[100], short ArrayLength) {
	short Min = Array[0];
	for (short  i = 1; i < ArrayLength; i++)
	{
		if (Array[i]< Min){
			Min = Array[i];
		}
	}
	return Min;
}
int main() {

	srand((unsigned)time(NULL));
	short Array[100], ArrayLength = 0;
	FillArrayWithRandomNumbers(Array, ArrayLength);
	cout << "\nArray Elements:";
	PrintArray(Array, ArrayLength);
	cout << "\nMin Number is : " << MinNumberInArray(Array, ArrayLength)<<endl;

	return 0;
}
