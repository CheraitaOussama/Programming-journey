#include<iostream>
using namespace std;

void FillArray(int Arr[10],int &ArrayLength) {
	ArrayLength = 6;
	Arr[0] = 10;
	Arr[1] = 20;
	Arr[2] = 30;
	Arr[3] = 30;
	Arr[4] = 20;
	Arr[5] = 10;
}
void PrintArray(int Arr[10], short ArrayLength) {
	for (int i = 0; i < ArrayLength; i++)
	{
		cout << Arr[i] << "   ";
	}
	cout << endl;
}
bool IsPalindromeArray(int Arr[10], short ArrayLength) {
	for (int  i = 0; i < ArrayLength; i++)
	{
		if (Arr[i] != Arr[ArrayLength - i-1])
			return false;
	}
	return true;
}

int main() {

	int Arr1[10], Array1Length = 0;
	FillArray(Arr1, Array1Length);
	cout << "\nArray 1 elements:\n";
	PrintArray(Arr1, Array1Length);
	if (IsPalindromeArray(Arr1, Array1Length)) {
		cout << "\nYes array is Palindrome.\n";
	}
	else
	{
		cout << "\nNo array is not Palindrome.\n";
	}
	

	return 0;
}
