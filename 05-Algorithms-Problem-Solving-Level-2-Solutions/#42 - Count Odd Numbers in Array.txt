#include<iostream>
#include<stdlib.h>
using namespace std;
enum enOddEven
{
	Odd=1, Even=2
};
enOddEven CheckNumber(int Number) {
	if (Number % 2 == 0)
		return enOddEven::Even;
	else
		return enOddEven::Odd;
}
int RandomNumber(int From,int To) {
	return rand() % (To - From + 1) + From;
}
void FillArrayWithRandomNumber(int Array[100], int &ArrayLength) {
	cout << "\nEnter Number of array elements\n";
	cin >> ArrayLength;
	for (int i = 0; i < ArrayLength; i++)
	{
		Array[i]=RandomNumber(1, 100);
	}
}
void PrintArray(int Array[100],int ArrayLength) {
	for (int i = 0; i < ArrayLength; i++)
	{
		cout << Array[i] << "    ";
	}
	cout << endl;
}
int CountOddNumbersInArray(int Array[100], int ArrayLength) {
	int Counter = 0;
	for (int  i = 0; i < ArrayLength; i++)
	{
		if (CheckNumber(Array[i]) == enOddEven::Odd) {
			Counter++;
		}
	}
	return Counter;
}

int main() {
	srand((unsigned)time(NULL));

	int Array[100],  ArrayLength=0;
	FillArrayWithRandomNumber(Array, ArrayLength);
	cout << "\nArray Elements: ";
	PrintArray(Array, ArrayLength);
	cout << "\nOdd Numbers count is : ";
	cout<<CountOddNumbersInArray(Array, ArrayLength)<<endl;

	return 0;
}
