#include<iostream>
using namespace std;
bool IsLeapYear(short Year) {
	return ((Year%400==0)|| (Year % 4 == 0 && Year % 100 != 0));
}
short NumberDaysInMonth(short Year, short Month){
	short DaysInMonth[12] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
	return (Month == 2) ? (IsLeapYear(Year)?29:28) : DaysInMonth[Month-1];
}
short ReadDay(short Year,short Month) {
	short Day = 0;
	do
	{
		cout << "\nPlease enter a Day?  ";
		cin >> Day;
	} while (Day<1 || Day > NumberDaysInMonth(Year, Month));
	return Day;
}
short ReadMonth() {
	short Month = 0;
	do
	{
		cout << "\nPlease enter a Month?  ";
		cin >> Month;
	} while (Month<1|| Month > 12);
	return Month;
}
short ReadYear() {
	short Year = 0;
	cout << "\nPlease enter a Year?  ";
	cin >> Year;
	return Year;
}
int GetDaysFromBeginingOfYear(short Day, short Month, short Year) {
	int TotalDays = 0;
	for (short i = 1; i <= Month-1; i++)
	{
		TotalDays += NumberDaysInMonth(Year, i);
	}
	TotalDays += Day;
	return TotalDays;
}
int main() {
	short Year = ReadYear();
	short Month = ReadMonth();
	short Day = ReadDay(Year, Month);

	cout << "Number of Days from the begining of the year is "<< GetDaysFromBeginingOfYear(Day, Month, Year);
	return 0;
}
