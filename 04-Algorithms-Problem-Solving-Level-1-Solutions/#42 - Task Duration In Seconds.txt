#include<iostream>
using namespace std;

struct strTaskDuration
{
	int Days, Hours, Minutes, Seconds;
};
int ReadPositiveNumber(string Message) {
	int Number = 0;
	cout << Message << endl;
	cin >> Number;
	return Number;
}

strTaskDuration ReadTaskDuration() {
	strTaskDuration TaskDuration;
	TaskDuration.Days = ReadPositiveNumber("Please enter Of Days? ");
	TaskDuration.Hours = ReadPositiveNumber("Please enter Of Hours? ");
	TaskDuration.Minutes = ReadPositiveNumber("Please enter Of Minutes? ");
	TaskDuration.Seconds = ReadPositiveNumber("Please enter Of Seconds? ");

	return TaskDuration;
}

int TaskDurationInSeconds(strTaskDuration TaskDuration) {
	int DurationInSeconds = 0;
	DurationInSeconds += TaskDuration.Days * 24 * 60 * 60;
	DurationInSeconds += TaskDuration.Hours * 60 * 60;
	DurationInSeconds += TaskDuration.Minutes  * 60;
	DurationInSeconds += TaskDuration.Seconds;

	return DurationInSeconds;
}

int main() {

	cout << "\nTask Duration In Seconds: " << TaskDurationInSeconds(ReadTaskDuration())<<" Seconds \n";

	return 0;
}
