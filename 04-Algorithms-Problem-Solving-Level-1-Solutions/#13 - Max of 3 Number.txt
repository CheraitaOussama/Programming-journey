int MaxOf3Numbers(int Number1, int Number2, int Number3) {
	if (Number1> Number2)
	{
		if (Number1>Number3)
			return Number1;
		else
			return Number3;
	}
	else
	{
		if (Number2 > Number3)
			return Number2;
		else
			return Number3;
	}
}
void PrintResult(int Max) {
	cout << "\nThe Maximum Number is : "<< Max <<endl;
}
