#include<iostream>
using namespace std;
// using reference variables.
void ReadNumbers(int &Num1, int& Num2, int& Num3) {
	cout << "\nPlease enter your number 1 ? " << endl;
	cin >> Num1;
	cout << "\nPlease enter your number 2 ? " << endl;
	cin >> Num2;
	cout << "\nPlease enter your number 3 ? " << endl;
	cin >> Num3;
}
int SumOf3Numbers(int Number1, int Number2, int Number3) {
	return Number1 + Number2 + Number3;
}
void PrintResults(int Total) {
	cout << "\nThe Sum of entered numbers is : " << Total << endl;
}

int main() {
	int Num1, Num2, Num3;
	ReadNumbers(Num1, Num2, Num3);
	PrintResults(SumOf3Numbers(Num1, Num2, Num3));
	return 0;
}
