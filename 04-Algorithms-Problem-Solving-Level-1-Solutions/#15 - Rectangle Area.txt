#include<iostream>
using namespace std;

void ReadNumber(float &A,float &B) {
	cout << "\nPlease Enter Rectangle Width A ?\n";
	cin >> A;
	cout << "\nPlease Rectangle Length B ?\n";
	cin >> B;
}
float CalculateRectangleArea(float A,float B) {
	return A * B;
}
void PrintResult(float A, float B) {
	cout << "\nRectangle Area = : "<< CalculateRectangleArea(A,B)<<endl;
}
int main() {
	float A, B;
	ReadNumber(A,B);
	PrintResult(A, B);
	return 0;
}
