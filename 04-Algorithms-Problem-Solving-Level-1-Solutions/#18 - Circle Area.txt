#include<iostream>
using namespace std;


float ReadRadius() {
	float R;
	cout << "\nPlease enter a Radius circle?\n";
	cin >> R;
	return R;
}
float CalculateCirlceArea(float R) {
	const float Pi = 3.14159265359;
	float Area = Pi * pow(R, 2);
	return Area;
}
void PrintResult(float Area) {
	cout << "\nCircle Area = "<< Area <<endl;
}
int main() {

	PrintResult(CalculateCirlceArea(ReadRadius()));
	return 0;
}
