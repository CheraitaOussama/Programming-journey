#include<iostream>
using namespace std;
int  ReadNumber() {
	int N;
	cout << "\nPlease Enter a Number?\n";
	cin >> N;
	return N;
}
void PrintRangeFromNto1_UsingDo_While(int Number) {
	int Counter = Number+1;
	cout << "Range printed using Do..While statement:\n";
	do
	{
		Counter--;
		cout << Counter << endl;
	} while (Counter>1);
}
void PrintRangeFromNTo1_UsingWhile(int Number) {
	int Counter = Number+1;
	cout << "Range printed using While statement:\n";
	while (Counter >1) {
		Counter--;
		cout << Counter << endl;
	}
}
void PrintRangeFromNto1_UsingFor(int Number) {
	cout << "Range printed using For Statement:\n";
	for (short Counter = Number; Counter >= 1; Counter--)
	{
		cout << Counter << endl;
	}
}
int main() {
	int N = ReadNumber();
	PrintRangeFromNto1_UsingDo_While(N);
	PrintRangeFromNTo1_UsingWhile(N);
	PrintRangeFromNto1_UsingFor(N);
	return 0;
}
