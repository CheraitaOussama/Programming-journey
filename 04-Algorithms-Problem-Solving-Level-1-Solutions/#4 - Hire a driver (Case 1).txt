#include<iostream>
using namespace std;

struct stInfo
{
	short Age;
	bool HasDrivingLisence;
};

stInfo ReadInfo() {
	stInfo Info;
	cout << "\nPlease Enter Your Age?"<<endl;
	cin >> Info.Age;
	cout << "\nDo you have a driver licence " << endl;
	cin >> Info.HasDrivingLisence;
	return Info;
}

bool IsAccepted(stInfo Info) {
	return (Info.Age > 21 && Info.HasDrivingLisence);
}

void PrintResult(stInfo Info) {
	if (IsAccepted(Info))
		cout << "\n Hired.\n";
	else
		cout << "\n Rejected.\n";
}

int main() {
	PrintResult(ReadInfo());
	return 0;
}
