#include<iostream>
#include<vector>
#include<cctype>
using namespace std;
vector<string> SplitStringInVector(string S1,string Delim) {
	vector<string>Vwords;
	short Position = 0;
	string Word;
	while ((Position=S1.find(Delim))!=string::npos) {
		Word = S1.substr(0, Position);
		if (Word!=""){
			Vwords.push_back(Word);
		}
		S1=	S1.erase(0, Position+ Delim.length());
	}
	if (S1 !=""){
		Vwords.push_back(S1);
	}
	return Vwords;
}
string JoinString(vector<string>Vwords, string Delim) {
	string S1;
	for (string& words : Vwords){

		S1 = S1 + words + Delim;
	}
	return S1.substr(0, S1.length()- Delim.length());
}
string LowerAllLetters(string S1) {
	for (short i = 0; i < S1.length(); i++)	{
		if (S1[i]!=' '&&isupper(S1[i]))
		S1[i] = tolower(S1[i]);
	}
	return S1;
}
string ReplaceWordInstringUsingSplit(string S1,string OldWord,string NewWord,bool matchcase=true) {
	vector<string> VWords = SplitStringInVector(S1," ");
	for (string& word : VWords)	{
		if (matchcase){
			if (word== OldWord){
				word = NewWord;
			}
		}
		else{
			if (LowerAllLetters(word) == LowerAllLetters(OldWord)) {
				word = NewWord;
			}
		}
	}
	return JoinString(VWords," ");
}


int main() {
	string S1 = "Welcome to Algeria , Algeria is a nice country";
	string StringToReplace = "algeria";
	string ReplaceTo = "USA";
	cout << "\nOriginal String:\n" << S1;
	cout << "\n\nReplace with match case: ";
	cout << "\n" << ReplaceWordInstringUsingSplit(S1,
		StringToReplace, ReplaceTo);
	cout << "\n\nReplace with dont match case: ";
	cout << "\n" << ReplaceWordInstringUsingSplit(S1,
		StringToReplace, ReplaceTo, false);
	return 0;
}
