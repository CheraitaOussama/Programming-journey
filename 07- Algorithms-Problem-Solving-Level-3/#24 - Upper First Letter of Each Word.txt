#include<iostream>
#include<cctype>
#include<string>
using namespace std;
string ReadString() {
	string Txt;
	cout << "Please Enter Your String?\n";
	getline(cin,Txt);
	return Txt;
}
string UpperFirstLetterOfEachWord(string S1) {
	bool IsFirstLetter = true;
	for (short i = 0; i < S1.length(); i++)
	{
		if (S1[i]!=' ' && IsFirstLetter && islower(S1[i]))
			S1[i] = toupper(S1[i]);
		IsFirstLetter = (S1[i] == ' ' ? true : false);
	}
	return S1;
}
int main() {
	string S1 = ReadString();
	cout << "String after conversion:\n";
	cout<<UpperFirstLetterOfEachWord(S1);
	return 0;
}
