vector<string> SplitString(string S1, string Seperator) {
	vector<string> Vwords;
	short Position = 0;
	string Word = "";
	while ((Position=S1.find(Seperator))!=string::npos) {
		Word = S1.substr(0, Position);
		if (Word!="")
		{
			Vwords.push_back(Word);
		}
		S1 = S1.erase(0, Position+ Seperator.length());
	}
	if (S1!="")	{
		Vwords.push_back(S1);
	}
	return Vwords;
}
string ConvertRecordToLine(StClientData Client,string Seperator="#//#") {
	string stClientRecord="";
	stClientRecord+=Client.AccountNumber+ Seperator;
	stClientRecord+= Client.PinCode+ Seperator;
	stClientRecord+=Client.Name + Seperator;
	stClientRecord+=Client.PhoneNumber+ Seperator;
	stClientRecord+=to_string(Client.AccountBalance);
	return stClientRecord;
}
StClientData ConvertLineToRecord(string Line, string Seperator = "#//#") {
	StClientData Client;
	vector<string> Vwords = SplitString(Line, Seperator);
	Client.AccountNumber=Vwords[0] ;
	Client.PinCode = Vwords[1];
	Client.Name = Vwords[2];
	Client.PhoneNumber = Vwords[3];
	Client.AccountBalance =stod(Vwords[4]);
	return Client;
}
void PrintClient(StClientData Client) {
	cout <<"Account Number : "<< Client.AccountNumber<<endl;

	cout <<"Name           : "<< Client.Name << endl;
	cout <<"Phone          : "<< Client.PhoneNumber << endl;
	cout <<"AccountBalance : "<< Client.AccountBalance << endl;
}
