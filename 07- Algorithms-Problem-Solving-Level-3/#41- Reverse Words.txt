#include<iostream>
#include<vector>
using namespace std;
vector<string> SplitStringInVector(string S1,string Delim) {
	vector<string>VWords;
	short Position = 0;
	string Word;
	while ((Position=S1.find(Delim))!=string::npos) {
		Word = S1.substr(0, Position);
		if (Word!=""){
			VWords.push_back(Word);
		}
		S1 = S1.erase(0, Position+ Delim.length());
	}
	if (S1!="")	{
		VWords.push_back(S1);
	}
	return VWords;
}
string ReverseWords(string S1, string Delim) {
	vector<string>VWords = SplitStringInVector(S1, Delim);
	string ReverseString;
	vector<string>::iterator it= VWords.end();
	while (it!= VWords.begin()) {
		it--;
		ReverseString = ReverseString + *it + Delim;
	}
	return ReverseString.substr(0, ReverseString.length()- Delim.length());
}
int main() {
	cout << ReverseWords("Oussama Cheraita I'm From Algeria", " ");

	return 0;
}
