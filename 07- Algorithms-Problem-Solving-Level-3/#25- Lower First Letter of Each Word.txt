#include<iostream>
#include<string>
#include<cctype>
using namespace std;
string ReadString() {
	string S1;
	cout << "\nPlease Enter Your String?\n";
	getline(cin,S1);
	return S1;
}
string LowerTheFirstLetterOfEachWord(string S1) {
	cout << "\nString after conversion:\n";
	bool IsFirstLetter = true;
	for (short i = 0; i < S1.length(); i++)
	{
		if (S1[i] != ' ' && IsFirstLetter && isupper(S1[i]))
			S1[i] = tolower(S1[i]);
		IsFirstLetter = ( S1[i] == ' ' ? true : false );
	}
	return S1;
}
int main() {
	string S1=ReadString();
	cout << LowerTheFirstLetterOfEachWord(S1);
	return 0;
}
