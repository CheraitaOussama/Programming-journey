#include<iostream>
#include<string>
using namespace std;

string ReadString() {
	string S1;
	cout << "\nPlease Enter your String?\n";
	getline(cin,S1);
	return S1;
}
void PrintEachWordInString(string S1) {
	short Position = 0;
	string Delim = " ";
	string Word;
	while ((Position= S1.find(Delim))!=string::npos) {
		Word = S1.substr(0, Position);
		if (Word!=" ")
		{
			cout << Word << endl; 
		}
		S1 = S1.erase(0, Position );
	}
	if (S1 != " ")
	{
		cout << S1 << endl;
	}
}
int main() {

	string S1 = ReadString();
	PrintEachWordInString(S1);
	return 0;
}
