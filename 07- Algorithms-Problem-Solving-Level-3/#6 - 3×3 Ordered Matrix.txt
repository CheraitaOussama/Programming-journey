#include<iostream>
#include<iomanip>
using namespace std;
void FillMatrixWithOrderedNumbers(int Matrix[3][3], short Row, short Column) {
	int Counter = 0;
	for (short i = 0; i < Row; i++)
	{
		for (short j = 0; j < Column; j++)
		{
			Matrix[i][j] = ++Counter;
		}
	}

}
void PrintMatrix(int Matrix[3][3], short Row,short Column) {
	for (short i = 0; i < Row; i++)
	{
		for (short j = 0; j < Column; j++)
		{
			cout << setw(3) << Matrix[i][j] << "   ";
		}
		cout << "\n";
	}
}

int main() {

	int Matrix[3][3];
	FillMatrixWithOrderedNumbers(Matrix,3,3);
	cout << "The following is a 3X3 ordered matrix:\n";
	PrintMatrix(Matrix,3,3);

	return 0;
}
