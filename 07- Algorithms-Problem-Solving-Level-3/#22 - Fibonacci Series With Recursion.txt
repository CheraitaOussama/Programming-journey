#include<iostream>
using namespace std;

void PrintFibonacci(short Number,int prev1,int prev2){
	int FibNumber=0;
	if (Number>0)
	{
		FibNumber = prev1 + prev2;
		prev2 = prev1;
		prev1 = FibNumber;
		cout << prev1 <<"     ";
		PrintFibonacci(Number - 1, prev1, prev2);
	}
}

int main(){
	PrintFibonacci(10,0,1);
	return 0;
}
