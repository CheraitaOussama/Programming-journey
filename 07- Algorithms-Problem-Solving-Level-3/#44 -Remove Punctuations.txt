#include<iostream>
#include<cctype>
using namespace std;
string RemoveAllpunctuation(string S1) {
	string S2;
	for (short i = 0; i < S1.length(); i++)
	{
		if (!ispunct(S1[i])){
			S2 += S1[i];
		}

	}
	return S2;
}
int main() {
	string S1 = "Welcome to Algeria, Algeria is a nice country; it's amazing.";
	cout << "Original String:\n";
	cout << S1 << endl;
	cout << "\nPunctuations Removed:\n";
	cout<<RemoveAllpunctuation(S1)<<endl;
	return 0;
}
